        function divide_in_group(~,candidates,subject, subject_code, gate)
            placeholder=candidates(strcmp(candidates.PSBAR1,subject)|  strcmp(candidates.PSBAR2,subject),:);
            placeholder_gen = placeholder(strcmp(placeholder.Category,"GEN") | strcmp(placeholder.Category,"GEN-EWS") ,:);
            if gate
                writetable(placeholder_gen,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_GEN_Gate"));
            else
                writetable(placeholder_gen,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_GEN_nongate"));
            end
            
            placeholder_obc = placeholder(strcmp(placeholder.Category,"OBC-NC") | strcmp(placeholder.Category,"OBC"),:);
            if gate
                writetable(placeholder_obc,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_OBC_Gate"));
            else
                writetable(placeholder_obc,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_OBC_nongate"));
            end
            
            placeholder_sc = placeholder(strcmp(placeholder.Category,"SC") | strcmp(placeholder.Category,"ST") ,:);
            if gate==1
                writetable(placeholder_sc,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_SC_Gate"));
            else
                writetable(placeholder_sc,"PhD_Candidate_latest.xlsx","Sheet",strcat(subject_code,"_SC_nongate"));
            end
        end

        function shortlisting(app,acronym,cutoff, pwd_cutoff,CFTI_cutoff, Non_CFTI_cutoff)
            table_data= readtable("PhD_Candidate_latest.xlsx","sheet",acronym);
            
            if(height(table_data)==0)
                writetable(table_data,"Scrutiny_Shortlist.xlsx","sheet",acronym);
                return;
            end
            masters_grades= table_data(~strcmp(table_data.QualifiengDegree,"B.Sc"),:); % Qualifying Degree shouldn't be B.SC
            masters_select= table_data(strcmp(table_data.QualifiengDegreeLevel,"Placeholder Degree ABCD"),:);
            x=height(masters_grades);
            for i=1:x
                hold=masters_grades(i,:);
                if strcmp(hold.PercentageOrCGPA,"CGPA")
                    temp=hold.QualiDegreeMaxMarks_PointScale;
                    temp=double(string(temp));
                    if(hold.MarksCGPAObtained>temp)
                        temp=100;
                    end
                    if strcmp("Yes",hold.PhysicalyDisabled)
                        if ((hold.MarksCGPAObtained * 100)/temp >=pwd_cutoff && checker(app,hold,CFTI_cutoff, Non_CFTI_cutoff))
                            masters_select(height(masters_select)+1,:)= hold;
                        end
                    elseif ((hold.MarksCGPAObtained * 100)/temp >=cutoff && checker(app,hold,CFTI_cutoff, Non_CFTI_cutoff))
                        masters_select(height(masters_select)+1,:)= hold;
                    end
                elseif (strcmp(hold.PercentageOrCGPA,"Percentage"))
                    if strcmp("Yes",hold.PhysicalyDisabled)
                        if (hold.MarksCGPAObtained >=pwd_cutoff && checker(app,hold,CFTI_cutoff, Non_CFTI_cutoff))
                            masters_select(height(masters_select)+1,:)= hold;
                        end
                    elseif (hold.MarksCGPAObtained >=cutoff && checker(app,hold,CFTI_cutoff, Non_CFTI_cutoff))
                        masters_select(height(masters_select)+1,:)= hold;
                    end
                end
            end
            writetable(masters_select,"Scrutiny_Shortlist.xlsx","sheet",acronym);
        end
        
        function [a]= checker(~,hold,CFTI_cutoff, Non_CFTI_cutoff)
            a=0;
            if strcmp(hold.CFTI,"Yes")
                if strcmp(hold.MSystem2,"Percentage")
                    if hold.MarksObtained2>=CFTI_cutoff
                        a=1;
                        return
                    end
                else
                    temp=double(string(hold.MaxMarks_Scale2));
                    if hold.MarksObtained2>temp
                        temp=100;
                    end
                    if (hold.MarksObtained2 * 100)/temp >=CFTI_cutoff
                        a=1;
                        return
                    end
                end
            else
                if strcmp(hold.MSystem2,"Percentage")
                    if hold.MarksObtained2>=Non_CFTI_cutoff
                        a=1;
                        return
                    end
                else
                    temp=double(string(hold.MaxMarks_Scale2));
                    if hold.MarksObtained2>temp
                        temp=100;
                    end
                    if (hold.MarksObtained2 * 100)/temp >=Non_CFTI_cutoff
                        a=1;
                        return
                    end
                end
            end
            
        end
        
        function Final_List(~,acronym,cutoff, gate,GEN_EWS_cutoff,PWD_cutoff)
            table_data=readtable("Scrutiny_Shortlist.xlsx","sheet",acronym);
            if height(table_data)==0
                writetable(table_data,"Final_List.xlsx","sheet",acronym);
                return;
            end
            if gate
                selected1= table_data( strcmp(table_data.PhysicalyDisabled,"No") & ~strcmp(table_data.Category,"GEN_EWS") & double(string(table_data.Score))>=cutoff ,:);
                selected2= table_data( strcmp(table_data.PhysicalyDisabled,"Yes") & double(string(table_data.Score))>=PWD_cutoff,:);
                selected3= table_data( strcmp(table_data.PhysicalyDisabled,"No") & strcmp(table_data.Category,"GEN_EWS") & double(string(table_data.Score))>=GEN_EWS_cutoff ,:);
                selected=[selected1 ;selected2 ;selected3];
                selected= sortrows(selected,"Score","descend");
